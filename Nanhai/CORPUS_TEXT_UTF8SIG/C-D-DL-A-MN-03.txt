མི་སྣ ངོ་སྤྲོད ། གཞོན་སྐྱེས བྲིས་པ སྨན་བླ་ཐར གྱི ཡིག་གཟུགས བདམས་བཀོད ། སྨན་བླ་ཐར ། སྨྱུག་མིང ལ མི་ཆུང  ། 1981ལོ-འི དགུན་ཟླ བཅུ་བ-ར མདོ་སྨད མང་རྫོང མང་ཆུ་སྔོན་མོ-འི འགྲམ་རྒྱུད ལོ་བ སྡེ་བ-ར ལྟེ་ཁྲག འཕོས ། 2002ལོ ནས ལོ་ངོ བཅུ་གཉིས ཀྱི རིང ལ རིམ་བཞིན མང་ར ས་ཆ-འི སྡེ་བ དང ཡུལ་ཚོ-འི སློབ་ཆུང དང རྫོང་མི་རིགས སློབ་འབྲིང སོགས སུ མི་རིགས སློབ་གསོ-འི ལས་དོན ལ རང་ནུས་ཅི་ལྕོགས ཀྱིས འབད་འབུངས བྱས ། 2011ལོ-ར ནུབ་བྱང མི་རིགས སློབ་ཆེན བོད སྐད་ཡིག་རིག་གནས སློབ་གླིང གི རྒྱ་བོད ལོ་ཙཱའི ཆེད་ལས ཀྱི རབ་འབྱམས ཞིབ་འཇུག སློབ་མ-འི བསླབ་གནས བླངས ། 2014ལོ-འི ཟླ ལྔ་བ-ར མཚོ་ལྷོ བོད་རིགས རང་སྐྱོང་ཁུལ བོད ཀྱི སྐད་ཡིག བྱ་བ-འི ཨུ་ཡོན ལྷན་ཁང གཞུང་ལས་ཁང གི བོད་ཡིག ཆ་འཕྲིན ལག་རྩལ ཞིབ་འཇུག ལྟེ་གནས སུ ལས་གནས སྤར ། སློབ་མ-འི དུས ནས ཡིག་གཟུགས འབྲི་སྦྱོང ལ དགའ་བ-ས དགེ་རྒན བཀའ་དྲིན ཅན ནམ་མཁའ་ཐར དང རིན་ཆེན་རྡོ་རྗེ གཉིས ཀྱི མདུན ནས དབུ་ཅན ཡིག་གཟུགས ཀྱི རྨང ལེགས་པ-ར བཏིངས་པ དང  ། རྒན མཚོ་སྐྱེས དོན་གྲུབ་ཚེ་རིང ལས དབུ་མེད སུག་རིང ལ སྦྱངས་པ གང་ལེགས བྱས ། གཞན གཙང གི བྲིས་པ གྲགས་ཅན ལྷ་ཁང བློ་བཟང་རྡོ་རྗེ མཆོག དང དབུས ཀྱི ཡིག་མཁན སྙེ འཕྲིན་ལས་རྒྱལ་མཚན མཆོག ཁམས ཀྱི མི་རྣམ་མཆོག་སྤྲུལ ། སྡེར་ནང ས་བྲིས ཀྱི རྒྱུད་འཛིན་པ རྒན ངག་དབང་རྒྱ་མཚོ སོགས ལས དབུས་གཙང གི དབུ་མེད ཀྱི བྲིས་རྒྱུན སྣ་ཚོགས དང སྙེ་བྲིས ། བཀའ་བརྒྱུད བྲབ་བྲིས ། ནང་བྲིས ནོར་བུ་མ ། སྡེར་ནང ས་བྲིས སོགས བྲིས་གཟུགས མི་འདྲ་བ-འི ཡིག་རྒྱུུགས ལ རང་བློ-ས དཔག་ཚོད ཀྱིས འབྲི་སྦྱོང གང་ནུས བྱས ། ཁོ་ནི མཚོ་ལྷོ་ཁུལ ཡིག་གཟུགས མཐུན་ཚོགས ཀྱི ཚོགས་མི ཡིན ཞིང  ། སྔ་ཕྱི-ར རྫོང དང ཁུལ ། ཞིང་ཆེན རིམ་པ-འི ཡིག་གཟུགས འགྲན་སྡུར ཁྲོད བྱ་དགའ རེ་འགའ ཡང ཐོབ མྱོང  ། 2015ལོ-ར གཟི་ཚ སྡེ་དགུ སྐྱེས་བུམ ཀྲུང་གོ ནུབ་རྒྱུད ཀྱི བོད་རྒྱ ཡིག་གཟུགས འགྲན་བསྡུར སྐབས དང་པོ-འི སྟེང ཟངས ཀྱི རྟགས་མ ཐོབ  ། ད་ལྟ མཚོ་ལྷོ་ཁུལ བོད་ཡིག ཆ་འཕྲིན ལག་རྩལ ཞིབ་འཇུག ལྟེ་གནས སུ མཚོ་སྔོན་པོ དྲ་བ དང ཡོངས་འཛིན བོད་ཡིག འཚོལ་བཤེར མ་ལག གི གྲངས་མཛོད རྩོམ་སྒྲིག དང རྒྱ་བོད ལོ་ཙཱའི ལས ལ འབད བཞིན ཡོད ལ ། དུས་ཚོད ཀྱི ཁོམ་གསེང དུ མུ་མཐུད ནས རྩོམ་འབྲི དང བོད ཀྱི ཡིག་གཟུགས སྒྱུ་རྩལ ལ བློ་སྦྱོང དང ལག་ལེན གང་ནུས བྱེད བཞིན མཆིས །
