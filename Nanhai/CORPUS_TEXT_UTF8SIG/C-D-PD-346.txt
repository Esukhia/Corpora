བེ་དོའི་སྡོང་པོ དང རྩྭ་འཇག་མ ། ༄༅ ། །གནའ་སྔ་མོ ནགས་ཚལ དཀྱིལ ལ རྒྱུག་ཆུ ཞིག ཡོད་པ དང དེའི འགྲམ དུ བེ་དོའི་སྡོང་པོ ཞིག སྐྱེས ཡོད ། ད་དུང དེའི ཉེ་འགྲམ དུ རྩྭ་འཇག་མ ཁ་ཤས ཀྱང སྐྱེས ཡོད ག་དུས ཡིན་ན འི རླུང་བུ ལྡང་བ འི སྐབས སྤོབས་པ མཐོ་བ འི བེ་དོའི་སྡོང་པོ དེ ས ཡལ་ག ཡོད་ཚད གནམ ལ གཟེངས ནས ཁ་ཐུག ལ ཡར་ལངས སྡོད་པ རེད ། འོན་ཀྱང རླུང་བུ གང་འདྲ ཞིག ལྡང་བ འི དུས གུས་གུས་ཞུམ་ཞུམ བྱེད་མཁན ནི རྩྭ་འཇག་མ ཚོ ཡིན་པ རེད ། ད་དུང རྩྭ་འཇག་མ ཚོས ཤུར་ཤུར གྱི གླུ་སྒྲ ཞིག ཀྱང སྒྲོག གི ཡོད་པ ས བེ་དོ འི བསམ་པ ར རྩྭ་འཇག་མ ཚོ རླུང་བུ རྒྱག་དུས ཞེད ནས ངུ གི འདུག སྙམ ། ཉིན་ཞིག བེ་དོའི་སྡོང་པོ ས རྩྭ་འཇག་མ ལ ། ཁྱེད་རང ལ དངོས་འབྲེལ ཐུགས་ཁག མི་འདུག རླུང་བུ བསིལ་བུ དེ་འདྲ ཞིག གིས ཁྱེད་རང མགོ་སྒུར དགོས་པ བཟོ་དུས གཟུགས་སྟོབས ཆུང་ཆུང ཟེར་བ དེ དངོས་འབྲེལ ཡ་ང དགོས་པ ཞིག རང་རེད ། ཅེས བཤད ། རྩྭ་འཇག་མ ས ཁྱེད་རང གིས ང་ཚོ ར སེམས་བཟང བྱས་པ དེ ར ཐུགས་རྗེ་ཆེ ། འོན་ཀྱང རླུང་བུ ས ང་ཚོ ར གནོད་འཚེ གང་ཡང བྱེད ཀྱི མི་འདུག གང་ཡིན་ཟེར་ན ང་ཚོ ས རླུང་བུ ར གུས་ཞབས བྱེད ཀྱི ཡོད ལ བསྟུན་མཁས བྱས ནས ཕན་ཚུན འཆམ་མཐུན གྱི འཚོ་བ སྐྱེལ གྱི ཡོད ། ད་དུང རླུང་བུ དང མཉམ་དུ རྩེད་མོ རྩེ ཡི ཡོད ། ང་ཚོ ས བྱས ན ཁྱེད་རང གི གནང་སྟངས དེ ཏོག་ཙམ འགྲིག གི མི་འདུག ཁྱེད་རང ལུས་སྟོབས རྒྱས་པོ དེ་འདྲ བྱུང་བ དེ ལ རླུང་བུ འི རམ་འདེགས མེད་པ ར ག་ལ ཡོང ཐུབ ། འོན་ཀྱང རླུང་བུ ཡོང་དུས བསྒུར་ཙམ ཡང མ་གནང་བ བྱས ། ཞེས བཤད ། ཡུན་རིང མ་སོང་གོང རླུང་བུ ཆེན་པོ ཞིག བརྒྱབ ། རྩྭ་འཇག་མ ཚོས གུས་གུས་ཞུམ་ཞུམ བྱས ཀྱང བེ་དོའི་སྡོང་པོ ས བསྒུར་ཙམ ཡང མ་བྱས་པ ར བརྟེན རླུང་བུ ཁོང་ཁྲོ ཟ ཐག་ཆོད ནས ནུས་ཤུགས ཡོད་ཚད ཀྱིས བརྒྱབ་པ ས བེ་དོའི་སྡོང་པོ དེ རྩ་བཀོག བཏང་པ རེད ། ནང་དོན ། གཞན ལ གུས་ཞབས བྱེད་པ དེ ནུས་ཤུགས མེད་པ འི རྟགས་མ ཡིན །
