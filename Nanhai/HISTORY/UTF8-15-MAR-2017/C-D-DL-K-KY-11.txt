མི་ཉག བུ་མོ ། ཨ་ལོང དངུལ གྱིས ཆེ་བ ། བཏང མ་ཐག ། བ་སོ དཀར ལ སྦོམ་པ-ས ། བཏང མ་ཐག ། བྱུར་དམར རྒྱན དུ བཏགས་པ ། བཏང མ་ཐག ། ཚེར་ཏོག དེ དང མཚུངས སོང། ། ཨ་གདན གདོང ལ འོངས་པ ། བཏང མ་ཐག ། ངོ་དཀར སྲུང་བ-འི ཉི་གདུགས ། བཏང མ་ཐག ། མིག དང སྨིན་མ-འི མཛེས་པ-ར ། བཏང མ་ཐག ། མདངས ཀྱི ཁ་དོག བསྩལ ཡོད ། བཏང མ་ཐག ། གསེར གྱི རྣ་ཆ གཟར་བ ། བཏང མ་ཐག ། མགྲིན ལ རྒྱན དུ འཕྱང་བ-ས ། བཏང མ་ཐག ། དམར་སེར ཚོན གྱི སྡོང་པོ ། བཏང མ་ཐག ། འགུལ ཙམ གཡོ ཙམ གནང འདུག ། སོ ལ གསེར་ཤུབས བསྐོན་པ ། བཏང མ་ཐག ། འགྲམ་ཟུང ཚོན་རྩི-ས སྨག་པ ། བཏང མ་ཐག ། ཤེལ་སྒོ-ར ལྟ བཞིན འགྲོ་བ ། བཏང མ་ཐག ། གཉུག་སྤྱོད གཙང་མ མངོན ནོ ། བཏང མ་ཐག ། སྟོད་ལྭ དམར་ཞད ཆེ ཞིང། ། སྐྲ་མཐུད དེ་ཡང དམར་བ-ས ། བཏང མ་ཐག ། དམར་དམར མཛེས་པ-འི སྒེག་ཉམས ། བཏང མ་ཐག ། ལང་ཚོ-འི ངོགས ནས གྲོལ ལོ ། བཏང མ་ཐག ། ཆབ་མ དངུལ གྱིས དཀར་བ ། བཏང མ་ཐག ། ཆེ་བ རྐེད་པ-འི སྐྱོར་ཡ ། བཏང མ་ཐག ། ཀྲོང་ངེར མཛེས་ཉམས མིན་ན ། བཏང མ་ཐག ། དགྱེ དང གུག་པ-ར དཀའ ཡོད ། བཏང མ་ཐག ། གློ་ཟུང ཁ་ལེབ འདྲ་བ ། བཏང མ་ཐག ། གཡས་གཡོན ཚར གྱིས དཔྱང བཞིན ། བཏང མ་ཐག ། ཁྲོག་སྒྲ སེང་སྒྲ བཅས་པ-ས ། བཏང མ་ཐག ། མིག གི ལྟད་མོ-ར བརྒྱན ཡོད ། བཏང མ་ཐག ། ཧྲིལ གྱིས དམར ཤུགས ཆེ ཞིང། ། ལྷམ གྱི ཕྱིར་རྟིང མཐོ་བ ། བཏང མ་ཐག ། བཅོས་མིན འཛུམ་ཉམས ཕྱུང སྟེ ། བཏང མ་ཐག ། མགྲོན ལ དགའ་བ-ས བསུས ཡོད ། བཏང མ་ཐག ། ངོ ལ དཀར་རྩི ཉུང ཞིང། ། སྨིན་མཆུ ཚོན་བྲིས མིན་པ-ས ། བཏང མ་ཐག ། མི་མོ ལྷད་མེད ཅིག གི ། རང་བཞིན ཉེ་བ-ར བཅར ཡོད ། བཏང མ་ཐག ། ཉི་ཟེར ལྷགས་རླུང བཟེད་པ-ས ། བཏང མ་ཐག ། སྐྱི་པགས འཇམ་ཤ བོར ཡང ། ། དགའ་དུང མཛའ་བརྩེ འཇམ་པོ ། བཏང མ་ཐག ། བྲང གི རུམ དུ བཅུག ཡོད ། བཏང མ་ཐག ། ལྷད་མེད སྒེག་འཕྱོར དེ ཡིས ། བཏང མ་ཐག ། གཞུང་བཟང ཕྱིར ལ མངོན བཞིན ། བཏང མ་ཐག ། རྒྱང་རིང མགྲོན་པོ སུ ལ ། བཏང མ་ཐག ། བསུ་མ-འི སེམས་སྐྱེས བཏེགས ཡོད ། །
