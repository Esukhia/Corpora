ཁྭ་ཏ་དང་ཝ་མོ ། གནའ་སྔ་མོ ཞིག ལ ། སྡོང་རྒན ཞིག གི མགོ་ན བྱ ཁྭ་ཏ ཞིག གིས ཚང བཞག ཡོད་པ དང བཏང མ་ཐག ། སྡོང་རྒན དེ འི རྩ་ན ཝ་མོ མ་བུ བཞི ས ཁུང་བུ བརྐོས ཏེ ཝ་ཚང བཅས ཡོད་པ རེད ། ཉིན་ཞིག བྱ ཁྭ་ཏ ས ཤ་རྡོག ཅིག སྡོང་མགོ རུ བཟུང འོང ནས ཟ བཞིན་པ འི སྐབས ། སྡོང་རྩ་ན ཡོད་པ འི ཝ་ཕྲུག རྣམས ཀྱིས མཐོང སྟེ ཟ འདོད ཀྱི འགྲམ་ཆུ ཁ ནས འཛར ཞོར དུ ། ཨ་ཁུ ཁྭ་ཏ ། ཨ་ཁུ ཁྭ་ཏ་ལགས ། ཉིད ཀྱིས ཟ་བཞིན་པ འི ཤ དེ ག་འདྲ འི ཞུམ མདོག ལ ། ང་ཚོ ར ཤ་རྡོག རེ སྤྲོད དང ཞེས ཞུ་བ འཐེན ། ཁྭ་ཏ ས ཤ་ལོག ཤིང་སྡོང གི ཡལ་ག འི སྟེང བཞག སྟེ རྫིགས ཉམས སྟོན་ཞོར དུ ན་རེ ། ཟས ཞིམ་པོ ཟ འདོད་ན ། རང གིས དཀའ་ལས རྒྱག དགོས རེད ཅེས མ་སྤྲད་པ རེད ། དགོང་མོ ཝ་མོ ཁྱིམ དུ བསླེབས དུས ། ཕྲུག་གུ རྣམས ཀྱིས ཁ་ཏ ས ཤ མ་སྤྲད་པ འི ལོ་རྒྱུས བཤད་པ་ན ། ཝ་མོ ས ང འི སྙིང གི གཅེས་ལོ་ཚོ ། ཉོན དང བཏང མ་ཐག ། འཇིག་རྟེན གྱི གཏམ་དཔེ ར ། དྲང་མོ ར བཤད་ན མི་དགའ ཟེར་བ ལྟར ། བྱ ཁྭ་ཏ ར ངོ་བསྟོད བཤད དགོས་པ ལས ། དྲང་མོ བཤད་ས ཡོད་པ མ་རེད ། སང་ཉིན ཁྱེད རྣམས ཀྱི ལྟོས དང བཏང མ་ཐག ། ང ས ཁ་ཏ འི ཁ ནས ཤ་རྡོག ཞིག ལེན ཐབས བྱེད ཅེས བཤད ནས ཉལ ། ཕྱི་ཉིན ཁྭ་ཏ ཡང ཤ་རྡོག ཞིག སྡོང་མགོ ར ཁྱེར ཡོང ནས ཟ གྲབས བྱེད་བཞིན་པ འི སྐབས སུ ། ཝ་མོ སྡོང་རྩ ར སོང སྟེ ། ངག འཇམ་པོ ས ། ཨ་ཁུ ཁྭ་ཏ་ལགས ། ཁྱོད ཀྱི ལུས ནི གསེར་ནག གི མདོག ཡིན་པ ས ། འཇིག་རྟེན འདི་ན ཁྱོད ལས མཛེས་པ ཞིག བྱུང མྱོང ཡོད་པ མ་རེད ། ཅེས ཁ་ཡག བཤད ། ཁྭ་ཏ ང་རྒྱལ དང བཅས སྡོང་མགོ ནས གཤོག་རྡེབ འགའ བྱས ནས བསྡད ། ཝ་མོ ས ཡང་བསྐྱར འདི་སྐད དུ ། ཨ་ཁུ ཁྭ་ཏ་ལགས ། འདབ་ཆགས གཞན་ཚོ ས བཤད་ན ། ཁྱོད བྱ ཀ་ལ་པང་ཀ འི ཚ་བོ རེད ཟེར ། ཁྱོད ལས ལྷག་པ འི ངག བདེ་བ དང བཏང མ་ཐག ། སྐད སྙན་པ ཞིག གནམ་འོག་ན ཡོད་པ མ་རེད ཟེར་བ ས ། ཁྱོད ཀྱིས གླུ ཞིག ལེན རོགས ཞེས ངོ་བསྟོད བྱས ། ཁྭ་ཏ སྔར ལས ཀྱང ང་རྒྱལ སྐྱེས་པ དང བཏང མ་ཐག ། སེམས ལ ། ང ས ཝ་ངན་མ འདི ལ གླུ ཞིག མ་བླངས་ན ། རྗེས སུ འདབ་ཆགས གཞན གྱིས འཕྱ་སྨོད བྱེད་རྒྱུ རེད བསམས ཏེ ། ཨ་བ ། ཨ་བ ། ཞེས ཁ གདེངས་པ དང མཉམ དུ ཁ འི ནང གི ཤ་ལོག མར ཟག སོང་བ རེད ། ཝ་མོ ས ཤ་ལོག ཁུར ཏེ དགོད་མ་འཛུམ གྱི ངང ནས ཁུང་བུ ར འཛུལ ནས ཝ་ཕྲུག་ཚོ ར ཤ ཟ རུ བཅུགས་སོ ། བཏང མ་ཐག །
