ཁོ འི ཁ་པར ཨང་གྲངས ང་ལ མེད ། ངག་དབང ལག་ན ཡོད རེད ། ངག་དབང རྩུབ་ལ ཅིག འགྲོ དགོ འདི སྐེ་ལ ཅིག བཏག རོགས བྱས ལབ སོང ། སུ ཡིས ། ད ཁོ འདི་ནས ནང་བསྟན་སྲུ་ཞུ་ཁང གི ལས་བྱེད ཅིག རེད འདུག་ག དེ་ནས ནང་བསྟན་སྲུ་ཞུ་ཁང གི ངོས་ནས འདི་ག རྫོང་སར་བཤད་གྲྭ ། སྒྲ་འཇུག བྱེད ཀྱི མངགས རེད ། སྒྲ་འཇུག ག་རེ རེད ཟེར་ན ། ད ཁོང་ཚོ འདི རེད འདུག་ག གཞི་རིམ གྱི སྐད་ཆ དེ འི ཐོག་ནས ལེན་ཡ ། ང་ཚོ ཡིས སྐད་ཆ བཤད་ཡ དང སློབ་ཚན དང དེབ ཀྱི ནང་ནས ། ད ཅིག བསྟན་བཅོས ཀྱི སྐད དང ཁ་སྐད ཟེར་ཡ ཅིག ཡོད རེད་པ ། དེ་ནས ཐག་རིང དྲག་ནས ། ད དེ འི དཀའ་ངལ དེ ཡོད རེད་པ ། ཨོ ད དེ མ་ཡོང་པ ། ད དེ ཐག་ཉེ་རུ གཏང་ནས ཟེར་ན མི་འདུག ད་ལྟ གཞི་རིམ གྱི སྐད དེ ཡིག་ཐོག ལ ཡོ ཡ དང ། རྩོམ་རིག གི ནང་ལ སླེབས་ཡ དེ་འདྲ འི ཁོ་ཚོ ས དེབ བསྒྲིག་ཡ ཡི འཆར་གཞི ཡོད་ས རེད་པ ། ནང་བསྟན་སྲུ་ཞུ་ཁང གིས ། ཨ་ནི ། དེ ཡོང་ཡ ཡི ཆེད་དུ སྒྲ་ལེན གྱི འདུག་ག ག་རེ ད རྒྱུན་ལྡན གྱི འཚོ་བ འི ནང གི སྐད་ཆ ། ད ཉིན་མ རེ་རེ་ལ སྐད་ཆ ག་རེ བཤད བསྡད་པ ཡིན་ན ། དེ འི ནང གི ཐ་སྙད སྟབས་བདེ་བོ དང ། ཁོ་རང ལས་སླ་བོ དང ། ཚིག་གྲུབ གང་ལྟར ཐ་སྙད བེད་སྤྱོད གཏང་སྟངས ཞིག ཡོད རེད་པ ། ཨོ ། དེ་ལ སྒྲ་འཇུག བཟོ ཡི འདུག་ག ཁོ་ཚོ ས རྡ་རམ་ས་ལ དེ་བར་ལ ཁ་ཤས བཟོས ཚར འདུག དེ་ནས འདི་ལ ཁ་སང ང་ཚོ ས མཁན་པོ ཞུས རེད་པ ། ང་ཚོ འི རྫོང ར གླིང གི དེ་ནས མཁན་པོ ཡིས གཞུང་འབྲེལ ཐོག་ལ ད ཅིག གྲྭ་བ གཏང་ཡ དེ་འདྲ ཡོང གི མ་རེད ། ག་རེ་ཟེར་ན ང་ཚོ ས ཅིག གཞུང་འབྲེལ ཐོག་ནས བོད་གཞུང་ནས ཅིག ཡོང་ན མ་གཏོགས ། རྩ་བ་ཉིད་ནས ཚོགས་པ དང མི་བུ སུ་ཞིག་ལ འང ། གཞུང་འབྲེལ ཐོག་ནས སྣེ་ལེན བྱེད ཀྱི མ་རེད ། དེ་ནས ཁྱེད་རང ད ཅིག འདི་ནས སློབ་མ དེ་ཚོ འི ནང་ནས སྒྲ་ཕབ་ན འགྲིག གི རེད ལབ སོང་ང ། དེ་ནས ད ཅིག ང ས བཏག བཞག ཡོད རེད ། ཚེ་རིང དཔལ་ལྡན་ལ ཅིག བཏག རོགས ལབ ཡིན ། ཁྱོད ཡང ཅིག བཏག རོགས བྱེད་གས ། འགྲིག གི རེད ། བཏག བཞག་ཡ གཅིག་པོ ། ནམ་རྒྱུན ང་ཚོ ས སྐད་ཆ བཤད་ཡ གཅིག་པོ དེ རེད་པ ། འདི་འདྲ བཏག བཞག་ན ཉིན་མ གཅིག གི སྒྲ འདི འི ནང་ལ འགྲོ བཞག གི རེད་པ ། རྡ་རམ་ས་ལ ཨ་ནེ ཅིག གི སྐེ་ལ བཏག བཞག འདུག་ག ཉིན་མ གཅིག་ལ སྐད་ཆ གཅིག་ར བཤད མི་འདུག ཟེར །
