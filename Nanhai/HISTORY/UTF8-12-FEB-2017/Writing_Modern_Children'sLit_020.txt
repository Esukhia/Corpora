རི་དྭགས ཤ་ར་བྷ ས རྒྱལ་བོ བསྐྱབས་པ ། དེ་ཡང སྔོན མི་མེད་ལུང་སྟོང གི དབེན་གནས ཤིག ཏུ བྱང་ཆུབ་སེམས་དཔའ བསམ་བཞིན དུ རི་དྭགས ཤ་ར་བྷ ཞིག ཏུ སྐྱེས ནས བཞུགས་སོ །  ། རི་དྭགས ཤ་ར་བྷ དེ ནི སོང་ན གོམ་ཁ ཡང ཞིང གཤིས་རྒྱུད ཧ་ཅང འཇམ་པ ཞིག ཡིན ། ཉིན ཞིག ལ ཡུལ དེ འི རྒྱལ་བོ རི་དྭགས གཤོར དུ འོངས་པ ས ། རྒྱང་རིང་བོ ནས རི་དྭགས ཤ་ར་བྷ དེ མཐོང ནས གསོད་པ འི་ཕྱིར རྟ་ཁ བསྐོར ཏེ རི་དྭགས དེ བདས ། རི་དྭགས ཤ་ར་བྷ ས གནས་ཚུལ དེ མཐོང རྗེས མྱུར་དུ བྲོས སོང ། རི་དྭགས དེ རབ་ཏུ མགྱོགས་པ འི ཤུགས ཀྱིས བྲོས་པ འི ལམ་ཁ་ན གྲོག་གཡང ཆེན་པོ ཞིག ཡོད་པ ལས མཆོངས ནས ཐར སོང ། རྒྱལ་བོ མི་རྟ ས རྗེས་བསྙེགས་པ ས གྲོག་གཡང དེ ལ མཆོངས མ་ནུས ཏེ རྟ་ཁ ཕྱི ར འཐེན་པ དང རྟ་ཐོག ནས གཡང དུ ལྷུང་ངོ ། རི་དྭགས སེམས་དཔའ་ཆེན་པོ དེ བྲོས་བྲོས་པ་ན རྟ འི རྨིག་སྒྲ མི་གྲགས་པ ཐོས ནས རྒྱལ་བོ སླར་ལོག སྙམ སྟེ ཕྱིར བལྟས་པ ས ། མི མེད་པར རྟ ཁོ་ན གྲོག་གཡང གི ཁ ན འདུག་པ མཐོང ། རི་དྭགས དེ འི སེམས་སུ རྒྱལ་བོ དེ གྲོག་གཡང དུ ལྷུང་བ ར གོར་མ་ཆག སྙམ ནས སྙིང་རྗེ ཆེན་པོ སྐྱེས ། དེ་ནས རི་དྭགས མཆོག སླར གྲོག་གཡང་ཁ ར སླེབས ནས མར གཟིགས་པ་ན ། རྒྱལ་བོ གྲོག ནང དུ ལྷུང སྟེ སྡུག་བསྔལ གྱིས མ་བཟོད་པ ར འགྲེ་ལྡོག བྱེད་པ མཐོང ། རི་དྭགས རུ་རུ ས སྙིང་རྗེ འི དབང གིས དགྲ འི འདུ་ཤེས བརྗེད དེ གྲོག་གཡང གི ཞབས ལ ཕྱིན ནས མིག མཆི ནས གང བཞིན རྒྱལ་བོར “ལུས ལ ཆག་གྲུམ མ་བྱུང ངམ ། བདག གིས ཁྱེད སྐྱབས ཁོ་ཐག ཡིན།” ཞེས གསུངས ། རྒྱལ་བོ ངོ་ཚ ནས དེ་སྔ འི བྱ་སྤྱོད ལ ཐུགས སྒོམ་པ ར ཞུས ། དེ་ནས རི་དྭགས ཤ་ར བྷས རྒྱལ་བོ ཁུར ནས ལེགས་པར འཇུས སུ བཅུག སྟེ གྲོག་ཁ ར ཐོན་པ ར བྱས ། ཁོ ས རྒྱལ་བོ བསྐྱབས རྗེས ཆིབས་རྟ འང བཙལ ཡོང ནས རྒྱལ་བོ ར གཏད ། རྒྱལ་བོ ཉེན་ཁ ལས ཐར རྗེས ཐུགས རབ་ཏུ དགའ ནས དད ཅིང གུས་པ ར རི་དྭགས་མཆོག དེ་རིང གི ཕོ་བྲང དུ གདན་དྲངས ཀྱང རི་དྭགས རུ་རུ ས དང་བླངས མ་མཛད ། ཞལ་གྱེས སྐབས ། རི་དྭགས རུ་རུ ས རྒྱལ་བོ ར སྡིག་སྤོང་དགེ་སྒྲུབ ཀྱི ཆོས བསྟན ནས ནགས་ཚལ སྟུག་པོ འི ཁྲོད དུ ཞུགས ཏེ སོང་ངོ། །
