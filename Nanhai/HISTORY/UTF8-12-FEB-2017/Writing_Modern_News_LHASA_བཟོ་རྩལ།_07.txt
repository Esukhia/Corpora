གྱན་ཐང ངོ་སྤྲོད ། རེབ་གོང སྒྱུ་རྩལ ལས གྱན་ཐང སྐོར ངོ་སྤྲོད མདོར་བསྡུས ། ཅི་ཞིག ལ གྱན་ཐང ཟེར ། གྱན་ཐང ཞེས བྱ་བ ནི དགོན་པ-འི ལྷ་ཁང དང འདུལ་ཁང གི གྱང་ངོས སུ བཞེངས ཡོད་པ-འི ལྡེབས་རིས ཆེན་པོ ལ ཟེར ཞིང  ། དེ་ནི པིར་བྲིས ཚོན་མདངས་ཅན གྱི རི་མོ-ར གཏོགས་པ ཡིན ། གྱན་ཐང གི ནང་དོན གཙོ་བོ ཅི་ཞིག ཡིན ། གྱན་ཐང གི ནང་དོན ནི སངས་རྒྱས དང བྱང་ཆུབ་སེམས་དཔའ སོ་སོ-འི སྐུ་བརྙན དང སངས་རྒྱས ཀྱི མཛད་པ ། རྣམ་ཐར བཅས དང  ། ལོ་རྒྱུས སྟེང གི སྐྱེས་བུ་དམ་པ རིམ་བྱོན གྱི རྣམ་ཐར ། བོད ཀྱི ནང་བསྟན ཆོས་ལུགས འཕེལ་རྒྱས ཀྱི ལོ་རྒྱུས སྟེང བྱུང་བ-འི གནས་ཚུལ གལ་པོ་ཆེ བཅས ཀྱི རྣམ་ཐར གཏམ་རྒྱུུད སོགས བྲིས ཡོད་པ རེད ། སངས་རྒྱས དང བྱང་ཆུབ་སེམས་དཔའ ། ཡི་དམ དཔའ་བོ མཁའ་འགྲོ ཆོས་སྐྱོང ལ་སོགས་པ-འི སྐུ་བརྙན རང་རང གི གོ་རིམ ལྟར བཞེངས ཡོད ། གྱན་ཐང བཞེངས་ཚུལ ། གྱན་ཐང བཞེངས་སྐབས ཐོག་མ-ར རས་གཞི-འི སྟེང སྐྱ་ཐིག རྒྱག་པ དང  ། དེ་ནས སྐྱ་ཐིག ནང་ལྟར ཚོན གཏོང་བ ། མཐར་གྱལ འཐེན་རྗེས བཞེངས ཚར་བ ཡིན ། དེ-ར རང གི ལག་རྩལ ལྟར རི་མོ དང ཚོན་མདངས གང་མང བཀོལ་ཆོག གྱན་ཐང གི ལོ་རྒྱུས ། ཆིན་རྒྱལ་རབས ཁང་ཞི དང ཆན་ལུང གཉིས ཀྱི རིང ལ ། གྱན་ཐང ཐད་གར གྱང་ངོས ལ བཞེངས་པ ཡིན ། ཁང་ཞི དང ཆན་ལུང གི རྗེས སུ ། རིམ་བཞིན རས་གཞི-འི སྟེང བཞེངས་རྗེས གྱང་ངོས ལ མཐའ ཤིང བཟོས ནས བཤམས ། དགོན་པ-འི ལྷ་ཁང ཕྱི་ནང བཅས ན གྱན་ཐང མང་པོ བཞེངས ཡོད་པ རེད ། གྱན་ཐང གི རིགས ། གྱན་ཐང ལ གསེར་ཤོག དང གསེར་བྲིས གང་མང ཡོད ། རེབ་གོང གྱན་ཐང གི ཁྱད་ཆོས ། རེབ་གོང སྒྱུ་རྩལ་པ ཚོས རས་གཞི བརྡར་བ ། ཚོན་སྣ སྡེབ་པ ། གསེར་ཤོག གི ལག་ལེན བྱེད་ཚུལ ། པིར་སྨྱུག ལས་ཚུལ བཅས ལ ཕྲ་ཞིང་ཞིབ་ལ ཐུན་མོང མ་ཡིན་པ-འི མན་ངག ཡོད ། པིར གྱིས བྲིས་པ-འི རི་མོ ཧ་ཅང ཆུང་བ མིག གིས མཐོང དཀའ་བ-འང འབྲི ཐུབ ། རེབ་གོང གྱན་ཐང ལ དུས་སྐབས སོ་སོ-ར འགྱུར་བ བྱུང་བ མང ན ཡང གྱན་ཐང གི ཕྲ་ཞིང་ཞིབ་པ-འི རི་མོ དང མཛེས་རྒྱན དང ཚོན་མདངས ཀྱི གཟི་མདངས ཀྱི རྣམ་པ དེ མ་དོར རྒྱུན་འཁྱོངས བྱས ཡོད་པ-ས བོད ཀྱི ནང་བསྟན ཆོས་ལུགས སྒྱུ་རྩལ གྱི ཁྲོད དུ འབུར དུ ཐོན་པ-འི ཁྱད་ཆོས ཡོད་པ དང  ། སངས་རྒྱས ཀྱི བསྟན་པ ལ དད་མོས བྱེད ས-འི ས་ཕྱོགས ཀུན ཏུ དར་ཁྱབ ཧ་ཅང ཆེ །
