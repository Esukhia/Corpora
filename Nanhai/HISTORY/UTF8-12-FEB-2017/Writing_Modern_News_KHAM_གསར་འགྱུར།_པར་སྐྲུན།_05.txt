གསར་འགྱུར ། ཏང གི སྐབས བཅོ་བརྒྱད་པ-འི ཀྲུང་ཨུ ཚང་འཛོམས གྲོས་ཚོགས ཐེངས བཞི་བ-འི དགོངས་དོན སློབ་སྦྱོང གི ཟུར་ཁྲིད རྒྱུ་ཆ པར དུ བསྐྲུན ། ཞིན་ཧྭ གསར་འགྱུར་ཁང གི གློག་འཕྲིན ལྟར ། རྒྱ་ཆེ-འི ཏང་ཡོན དང ལས་བྱེད་པ ། མང་ཚོགས བཅས ཀྱིས ཏང གི སྐབས བཅོ་བརྒྱད་པ-འི ཀྲུང་ཨུ ཚང་འཛོམས གྲོས་ཚོགས ཐེངས བཞི་བ-འི དགོངས་དོན ལ སློབ་སྦྱོང དང དེ ལག་བསྟར ཟབ་མོ བྱེད་པ-ར རོགས་རམ བྱེད ཆེད ། ཀྲུང་དབྱང འབྲེལ་ཡོད སྡེ་ཁག གིས རྩ་འཛུགས བྱས ཏེ ཀྲུང་གུང ཀྲུང་དབྱང གི ཁྲིམས ལྟར རྒྱལ་ཁབ སྐྱོང་བ-ར ཕྱོགས ཡོངས ནས སྐུལ་འདེད བྱེད་པ-འི གནད་དོན གལ་ཆེན རབ་དང་རིམ་པ-འི སྐོར གྱི ཆོད་དོན གྱི ཟུར་ཁྲིད ཀློག་དེབ དང ། ཏང གི སྐབས བཅོ་བརྒྱད་པ-འི ཀྲུང་ཨུ ཚང་འཛོམས གྲོས་ཚོགས ཐེངས བཞི་བ-འི ཆོད་དོན སློབ་སྦྱོང ཟུར་ཁྲིད ཀྱི དྲི་བ བརྒྱ་ཐམ་པ ཞེས་པ-འི དཔེ་དེབ རྩོམ་སྒྲིག བྱས ནས ། ཚང་འཛོམས གྲོས་ཚོགས ཀྱི ཆོད་དོན ལ ཕྱོགས ཡོངས ནས འགྲེལ་བཤད བྱས་པ དང །དེ་ནི ཚང་འཛོམས གྲོས་ཚོགས ཀྱི དགོངས་དོན ལ སློབ་སྦྱོང བྱས ཏེ གོ་བ ལེན་པ-འི དབང་གྲགས་ཅན གྱི ཟུར་ཁྲིད རྒྱུ་ཆ རེད ། དཔེ་དེབ ནང དུ སྤྱི་ཁྱབ ཧྲུའུ་ཅི ཞི་ཅིན་ཕེན གྱིས ཏང གི སྐབས བཅོ་བརྒྱད་པ-འི ཀྲུང་ཨུ ཚང་འཛོམས གྲོས་ཚོགས ཐེངས བཞི་བ-འི སྟེང གནང་བ-འི ཀྲུང་གུང ཀྲུང་དབྱང གི ཁྲིམས ལྟར རྒྱལ་ཁབ སྐྱོང་བ-ར ཕྱོགས ཡོངས ནས སྐུལ་འདེད བྱེད་པ-འི གནད་དོན གལ་ཆེན རབ་དང་རིམ་པ-འི ཆོད་དོན སྐོར གྱི གསལ་བཤད བཏུས ཡོད ། ཉེ་ལམ ཀྲུང་གུང ཀྲུང་དབྱང གི ཁྲིམས ལྟར རྒྱལ་ཁབ སྐྱོང་བ-ར ཕྱོགས ཡོངས ནས སྐུལ་འདེད བྱེད་པ-འི གནད་དོན གལ་ཆེན རབ་དང་རིམ་པ-འི སྐོར གྱི ཆོད་དོན གྱི ཟུར་ཁྲིད ཀློག་དེབ དེ མི་དམངས དཔེ་སྐྲུན་ཁང གིས པར་སྐྲུན འགྲེམ་སྤེལ བྱས་པ དང །ཏང གི སྐབས བཅོ་བརྒྱད་པ-འི ཀྲུང་ཨུ ཚང་འཛོམས གྲོས་ཚོགས ཐེངས བཞི་བ-འི སློབ་སྦྱོང ཟུར་ཁྲིད ཀྱི དྲི་བ བརྒྱ་ཐམ་པ ཞེས་པ-འི དཔེ་དེབ དེ ཏང་འཛུགས དཔེ་སྐྲུན་ཁང དང སློབ་སྦྱོང དཔེ་སྐྲུན་ཁང གིས པར་སྐྲུན འགྲེམ་སྤེལ བྱས ། །
