ར་མ དང ཝ་མོ སྤྱང་ཀི གནའ་སྔ་མོ ཞིག ལ བོད་ཡུལ་ན ཀུན ཏུ ར་བ ཞེས འབོད་པ འི ཡུལ་གྲུ ཞིག ཡོད་པ དེ རུ ། མཛེས་སྡུག་ལྡན་པ འི རྩྭ་ཐང དང དལ གྱིས བཞུར་བ འི གཙང་བོ ཞིག ཡོད ། ཉིན་ཞིག ལ འདོད་སྲེད་ཅན གྱི སྤྱང་ཀི ཞིག རྩྭ་ཐང་ན ཁེར་རྐྱང དུ གཟན་ཆས འཚོལ གྱིན འདུག སྤྱང་ཀི ས ར་མ ཞིག གིས གཙང་འགྲམ དུ རྩྭ ཟ བཞིན འདུག་པ མཐོང མོད ། འོན་ཀྱང ར་མར ངར་ཅན གྱི གྲོགས་པོ ཁྱི་རྒན་རྒྱ་བོ ཞིག ཡོད་པ ས ། སྤྱང་ཀི དེ བཀྲེས་ལྟོགས ཀྱི ངང ཕྱིར མི་ལོག ཐབས་མེད བྱུང  ། ཕྱི་ཉིན སྤྱང་ཀི དེ ལམ་བར ནས ཝ་མོ ཞིག དང འཕྲད ། སྤྱང་ཀི ས“ སྙིང་ཉེ་བ འི ཨ་ཅེ ཝ་མོ ལགས ། ཁྱེད ནི འཛམ་གླིང སྲོག་ཆགས ཁྲོད དུ ཆེས བློ་རིག ལྡན་པ དེ ཡིན ། ཁྱེད ཀྱིས ང ལ རོགས བྱས ཏེ ར་མ དང ཁྱི་རྒན་རྒྱ་བོ དེ་གཉིས སོ་སོ ར བཀར ཐུབ་ན ། འུ་གཉིས ཀྱིས ར་མ དེ མཉམ་ཟ བྱས་ན ཆོག” ཅེས བཤད་པ ས ། ཝ་མོ དེ འང འཐད་པ བྱུང སྟེ ར་མ ར འཚམས་འདྲི བྱེད དུ སོང  ། ཝ་མོ ས ངག འཇམ་པོ འི ངང བཤད་རྒྱུ ར ། “ཀྭ་ཡེ ། སྙིང་ཉེ་བ འི ར་མ རྒན་མོ་ལགས ། ཁྱེད ནི བརྩེ་སེམས ཆེ་བ འི རྩྭ་གཟན ཞིག ཡིན ། འོན་ཀྱང ཁྱེད ཀྱི གྲོགས་པོ ཁྱི་རྒན་རྒྱ་བོ དེ་ནི གདུག་རྩུབ ཆེ ཞིང སྙིང་རྗེ་མེད་པ ས ། ཉིན་ཞིག དེ ས ཁྱོད་རང ཁ་ཟས སུ བྱེད ཀྱང སྲིད ། དེ་བས ཁྱེད ཀྱི ས ངེས་པར་དུ སེམས་ཆུང བྱེད རོགས” ཞེས བཤད ། ར་མ འི སེམས སུ ཤ་ཚ་བ འི གཏམ ལ སྙན་མོ་མེད ཅེས་པ འི ཁ་དཔེ དེ དྲན ཅིང  ། ཝ་མོ ཡིས ཕན ཆེ་བ འི བསླབ་བྱ གནང་བ ར བཀའ་དྲིན ཞུས ། ཕྱི་ཉིན ར་མ ས ཁྱི་རྒན ལ“ ཝ་ཡེ ཁྱི་རྒན་རྒྱ་བོ ལགས ། ཁྱོད ནི ང འི གྲོགས་པོ ཡིན མོད འོན་ཀྱང མི་ཚང་མ ས བཤད་ན གཏུམ་དྲག་ཅན ཁྱོད ཀྱིས ཉིན་ཞིག ང་རང ཡང ཟ འདོད ཡོད་པ ར གསུངས་པ ས ། ཁྱེད་རང འདི རུ མ་འདུག གནས གཞན ཞིག ལ འགྲོ་རོགས གནང” ཞེས བཤད་པ ས ། ཁྱི་རྒན་ངར་ཅན དེ ཡུལ་གཞན ཞིག ཏུ སོང་ངོ ། ། ཕྱི་ཉིན སྔ་མོ ར སྤྱང་ཀི ཡོང ནས ར་མ བསད ཅིང ། རྫུན་གཏམ ལ ཆེས་པ འི ར་མ རྒན་མོ དེ ཁོ་གཉིས ཀྱི ཞིམ་མངར་ལྡན་པ འི ཟས་ཐུན གཅིག ཏུ གྱུར་ཏོ ། །
